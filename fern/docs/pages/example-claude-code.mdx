This walkthrough shows a complete Claude Code session where the user asks the agent to sell a file. The agent handles account setup, listing creation, and readiness resolution autonomously — asking the user only when human action is required.

---

## The scenario

The user has an ebook PDF in their S3 bucket and wants to sell it for $19. They have not set up a ListBee account yet. They do have a Stripe account.

---

## Step 1: Check for an existing API key

The agent checks the environment for a ListBee API key before doing anything else.

```python
import os
import httpx

BASE = "https://api.listbee.so"

api_key = os.environ.get("LISTBEE_API_KEY")

if api_key:
    # Verify the key is valid
    resp = httpx.get(
        f"{BASE}/v1/account",
        headers={"Authorization": f"Bearer {api_key}"},
    )
    if resp.status_code == 200:
        print("API key found and valid.")
        account = resp.json()
    else:
        print("API key found but invalid. Will create a new account.")
        api_key = None
else:
    print("No LISTBEE_API_KEY found. Starting account setup.")
```

**Agent says to user:**
> I don't see a ListBee API key in your environment. I'll create an account for you. What's your email address?

**User says:**
> alex@example.com

The next action is `kind: "api"` — the agent can handle it without a browser.

**Agent says:**

> sk_live_51Abc123...

```python

    f"{BASE}/v1/account/stripe-key",
    headers=headers,
```

## Step 4: Create the listing

```python
import uuid

resp = httpx.post(
    f"{BASE}/v1/listings",
    headers={**headers, "Idempotency-Key": str(uuid.uuid4())},
    json={
        "name": "Python Async Patterns",
        "price": 1900,      # $19.00
        "currency": "usd",
        "content": "https://alex-bucket.s3.amazonaws.com/python-async.pdf",
    },
)
listing = resp.json()

print(listing["url"])                   # https://buy.listbee.so/python-async-patterns
print(listing["readiness"]["sellable"]) # True
```

## Step 5: Check listing readiness

```python
if listing["readiness"]["sellable"]:
    print(f"Listing is live: {listing['url']}")
else:
    # Handle any remaining blockers
    next_code = listing["readiness"]["next"]
    action = next(
        a for a in listing["readiness"]["actions"]
        if a["code"] == next_code
    )
    if action["kind"] == "api":
        httpx.request(
            action["resolve"]["method"],
            f"{BASE}{action['resolve']['endpoint']}",
            headers=headers,
            json=action["resolve"]["params"],
        )
    else:
        print(f"Action required: {action['message']}")
        print(f"Visit: {action['resolve']['url']}")
```

**Agent says to user:**
> Your ebook is live. Share this URL with buyers:
>
> **https://buy.listbee.so/python-async-patterns**

## Full script

```python
import os
import uuid
import httpx

BASE = "https://api.listbee.so"


def get_or_create_account() -> tuple[str, dict]:
    """Return (api_key, account) — create account if no key in env."""
    api_key = os.environ.get("LISTBEE_API_KEY")

    if api_key:
        resp = httpx.get(f"{BASE}/v1/account", headers={"Authorization": f"Bearer {api_key}"})
        if resp.is_success:
            return api_key, resp.json()

    email = input("ListBee email: ").strip()
    httpx.post(f"{BASE}/v1/account/signup", json={"email": email})

    code = input("Verification code from inbox: ").strip()
    data = httpx.post(
        f"{BASE}/v1/account/verify",
        json={"email": email, "code": code},
    ).json()

    api_key = data["api_key"]
    print(f"\nSave this: export LISTBEE_API_KEY={api_key}\n")
    return api_key, data["account"]


def ensure_stripe(api_key: str, account: dict) -> None:
    """Submit Stripe key if not connected."""
    if account["readiness"]["operational"]:
        return

    headers = {"Authorization": f"Bearer {api_key}"}
    next_code = account["readiness"]["next"]

    if next_code == "set_stripe_key":
        stripe_key = input("Stripe secret key (sk_live_...): ").strip()
        httpx.post(
            f"{BASE}/v1/account/stripe-key",
            headers=headers,
            json={"secret_key": stripe_key},
        )
    elif next_code == "connect_stripe":
        url = httpx.post(f"{BASE}/v1/account/stripe/connect", headers=headers).json()["url"]
        print(f"Connect Stripe: {url}")
        input("Press Enter when done...")


def create_listing(api_key: str, name: str, price_cents: int, content_url: str) -> dict:
    """Create listing and return response."""
    return httpx.post(
        f"{BASE}/v1/listings",
        headers={
            "Authorization": f"Bearer {api_key}",
            "Idempotency-Key": str(uuid.uuid4()),
        },
        json={
            "name": name,
            "price": price_cents,
            "currency": "usd",
            "content": content_url,
        },
    ).json()


# Main
api_key, account = get_or_create_account()
ensure_stripe(api_key, account)

listing = create_listing(
    api_key,
    name="Python Async Patterns",
    price_cents=1900,
    content_url="https://alex-bucket.s3.amazonaws.com/python-async.pdf",
)

if listing["readiness"]["sellable"]:
    print(f"\nListing live: {listing['url']}")
else:
    print(f"\nNot yet sellable. Next action: {listing['readiness']['next']}")
```
